import numdifftools as nd
import numpy as np
from numpy.linalg import norm, inv, cholesky,LinAlgError

def testfunction(x): 
    return (1-x[0])**2 + 100*(x[0]-x[1])**2
# return (1-x[0])**2 + 105.*(x[1]-x[0]**2)**2
def qs(s,f,g,G): 
    return f + np.dot(g,s) + 0.5*np.dot(np.dot(s,G),s)
